PackageAttribute

Description

This attribute provides a more details information about a namepace package module in your scripting plugins.

Declare

            
# namespace Microsoft.VisualBasic.Scripting.MetaData
export class PackageAttribute {
   # This plugins project's home page url.
   Url: string;
   # Your name or E-Mail
   Publisher: string;
   Revision: integer;
   # 这个脚本模块包的文献引用列表
   Cites: string;
   Category: APICategories;
   Description: string;
   Namespace: string;
   AutoExtract: boolean;
   TypeId: any kind;
}
