MetaboliteAnnotation

Description

A basic metabolite annotation data model

contains the basic annotation metadata: id, name, exact mass, and formula data

# namespace BioNovoGene.BioDeep.Chemoinformatics
export class MetaboliteAnnotation {
   # common name of the current metabolite
   CommonName: string;
   # exact mass value of current metabolite, which it could be evaluated from the MetaboliteAnnotation.Formula
   ExactMass: double;
   # atom compositions of current metabolite
   Formula: string;
   # the unique reference id of current metabolite
   Id: string;
}
