export class SingleScanInfo {

   CollisionMode: string;

   EventNumber: integer;
   # Scan Filter string
   FilterText: string;

   Frequency: double;
   # Highest m/z value
   HighMass: double;

   IonInjectionTime: double;
   # Ionization polarity
   IonMode: IonModes;
   # Centroid scan flag
   IsCentroided: boolean;
   # FTMS flag (or Orbitrap, Q-Exactive, Lumos, or any other high resolution instrument)
   IsFTMS: boolean;
   # The monoisotopic parent ion m/z, as determined by the Thermo software
   IsolationWindowTargetMZ: double;
   # Lowest m/z value
   LowMass: double;
   # MRM mode
   MRMInfo: MRMInfo;
   # Multiple reaction monitoring mode
   MRMScanType: MRMScanTypeConstants;
   MSData: RawLabelData;
   # MS Level
   MSLevel: integer;
   # Number of channels
   NumChannels: integer;
   # Number of mass intensity value pairs
   NumPeaks: integer;
   # The monoisotopic parent ion m/z, as determined by the Thermo software
   ParentIonMonoisotopicMZ: double;
   # Parent ion m/z
   ParentIonMZ: double;
   # Retention time (in minutes)
   RetentionTime: double;
   # Scan event data
   ScanEvents: tuple[string, string];
   # Scan number
   ScanNumber: integer;
   # SIM Scan Flag
   SIMScan: boolean;
   # Status log data
   StatusLog: tuple[string, string];
   # Total ion current
   TotalIonCurrent: double;
   # Indicates whether the sampling time increment for the controller is constant
   UniformTime: boolean;
   # Zoom scan flag
   ZoomScan: boolean;
}
